# Function to plot line graphs with values annotated
def plot_line_graph_with_values(df, value_col, title, ylabel):
    pivot_df = df.pivot(index='year', columns='clm_line_hcpcs_cd', values=value_col)
    ax = pivot_df.plot(kind='line', marker='o', figsize=(10, 6))

    # Annotating values on the graph
    for line, product in zip(ax.get_lines(), pivot_df.columns):
        for x, y in zip(pivot_df.index, pivot_df[product]):
            ax.annotate(f"{round(y)}", (x, y), textcoords="offset points", xytext=(0, 5), ha='center', fontsize=8)

    plt.title(title)
    plt.xlabel('Year')
    plt.ylabel(ylabel)
    plt.legend(title='Product', bbox_to_anchor=(1.05, 1), loc='upper left')
    plt.xticks(ticks=pivot_df.index, labels=pivot_df.index)  # Ensure x-axis only shows integer years
    plt.grid(True, linestyle='--', alpha=0.7)
    plt.tight_layout()
    plt.show()

# Plot claims per HCP
