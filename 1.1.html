import pandas as pd
import matplotlib.pyplot as plt

# Load and preprocess data
data = pd.read_csv('merged_data_meghana.csv')  # Your data file
data['clm_from_dt'] = pd.to_datetime(data['clm_from_dt'], errors='coerce')
data['year'] = data['clm_from_dt'].dt.year

# Filter for the four products
products = ['J3010', 'J1885', 'J2250', 'J2704']
filtered_data = data[data['clm_line_hcpcs_cd'].isin(products)]

filtered_data = filtered_data[
    (filtered_data['clm_from_dt'].dt.year >= 2016) &
    (filtered_data['clm_from_dt'].dt.year <= 2018)]

# Calculate metrics
metrics = filtered_data.groupby(['year', 'clm_line_hcpcs_cd']).agg(
    claims=('cur_clm_uniq_id', 'count'),
    patients=('bene_mbi_id', 'nunique'),
    hcps=('fac_prvdr_npi_num', 'nunique')
).reset_index()

# Normalize to percentages for stacked bar charts
metrics['claims_percentage'] = metrics.groupby('year')['claims'].transform(lambda x: x / x.sum() * 100)
metrics['patients_percentage'] = metrics.groupby('year')['patients'].transform(lambda x: x / x.sum() * 100)
metrics['hcps_percentage'] = metrics.groupby('year')['hcps'].transform(lambda x: x / x.sum() * 100)

# Function to plot 100% stacked bar chart with numbers
def plot_stacked_bar_with_numbers(df, value_col, title, ylabel):
    pivot_df = df.pivot(index='year', columns='clm_line_hcpcs_cd', values=value_col)
    ax = pivot_df.plot(kind='bar', stacked=True, figsize=(10, 6), colormap='tab10')
    
    # Adding numbers on top of bars
    for i, bars in enumerate(ax.containers):
        for bar in bars:
            height = bar.get_height()
            if height > 0:  # Only annotate non-zero bars
                ax.text(
                    bar.get_x() + bar.get_width() / 2,  # X-coordinate (center of the bar)
                    bar.get_y() + height / 2,  # Y-coordinate (middle of the bar)
                    f'{height:.1f}%',  # Text (formatted to 1 decimal place)
                    ha='center', va='center', fontsize=8, color='black'
                )

    plt.title(title)
    plt.xlabel('Year')
    plt.ylabel(ylabel)
    plt.legend(title='Product', bbox_to_anchor=(1.05, 1), loc='upper left')
    plt.tight_layout()
    plt.show()
